Lead Design

M = 1000;

Lead_num = [1, wc/sqrt(M)];
Lead_den = [1, wc*sqrt(M)];
Leadtf = M*tf(Lead_num,Lead_den);

PC_mag_wr = bode(Phtf*Leadtf,wr);
LeadK = PC_mag_des/PC_mag_wr;

margin(Phtf*Leadtf*LeadK)
legend('Plant','Lead')

PC_mag_wr = bode(Phtf*Leadtf*LeadK,wr);
